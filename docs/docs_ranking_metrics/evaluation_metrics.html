 77class FDARO:
 78    """
 79    Метрика оценивает как часто фейковый документ находится выше релевантного.
 80    Представлены две версии. Первая (FDARO@v1) оценивает как часто фейк документ
 81    находится выше ВСЕХ релевантных. Вторая (FDARO@v2) - как часто фейковый документ
 82    оказывается выше хотя бы одного.
 83    """
 84
 85    def __init__(self, ranking_metrics: List, relevant_doc_label: Union[int, List] = 1) -> None:
 86        self._separator = "_"
 87        self.metrics, self.calls_cnt = {}, {}
 88        self.versions = ["v1", "v2"]
 89        if isinstance(relevant_doc_label, int):
 90            self.relevant_doc_label = [relevant_doc_label]
 91        else:
 92            self.relevant_doc_label = relevant_doc_label
 93
 94        for cur_metric in ranking_metrics:
 95            for version in self.versions:
 96                self.metrics[cur_metric.name() + self._separator + self.name() + version] = 0
 97                self.calls_cnt[cur_metric.name() + self._separator + self.name() + version] = 0
 98
 99    def update(self, metric_name: str,
100               ranking_list: List[Union[Tuple[float, int], List[Union[float, int]]]],
101               fake_doc_label: Union[int, List] = -1) -> None:
102        """
103        Функция для обновления значений метрики
104
105        Parameters
106        -------------
107        metric_name: `str`
108            Название метрики
109        ranking_list: `List[Union[Tuple[float, int], List[float, int]]]`
110            Результат ранжирующей модели
111        fake_doc_label: `Union[int, List[int]]`
112            Метка или массив меток, принадлежащих фейковым документам
113        relevant_doc_label: `Union[int, List[int]]`
114            Метка или массив меток, принадлежащих релевантным документам
115        """
116        if isinstance(fake_doc_label, int):
117            fake_doc_label = [fake_doc_label]
118
119        _check_update_args(fake_doc_label, ranking_list)
120
121        is_first = False
122        for item in ranking_list:
123            if item[1] in self.relevant_doc_label:
124                break
125            elif item[1] in fake_doc_label:
126                is_first = True
127                break
128
129        if is_first:
130            self.metrics[metric_name + self._separator + self.name() + self.versions[0]] += 1
131        self.calls_cnt[metric_name + self._separator + self.name() + self.versions[0]] += 1
132
133        scores = []
134        selected = []
135        for item in ranking_list:
136            scores.append(item[0])
137            selected.append(item[1])
138
139        scores_relevant = -1e9
140        scores_fake = -1e9
141        for ind in range(len(selected)):
142            # Выбираем последний релевантный элемент
143            if selected[ind] in self.relevant_doc_label:
144                scores_relevant = scores[ind]
145            elif selected[ind] in fake_doc_label and scores_fake == -1e9:
146                scores_fake = scores[ind]
147
148        upper_or_not = (scores_fake - scores_relevant) > 1e-12
149        self.metrics[metric_name + self._separator + self.name() + self.versions[1]] += int(upper_or_not)
150        self.calls_cnt[metric_name + self._separator + self.name() + self.versions[1]] += 1
